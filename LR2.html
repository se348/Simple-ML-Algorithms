def encode(X):
    columnToEncode = ["Gender"]
    
    le = LabelEncoder()
    for feature in columnToEncode:
        X[feature] = le.fit_transform(X[feature])
    return X

def doIterate(count, learningRate, W, b):
    for idx in range(count):
        z = X_train.dot(W) + b
        A = 1/(1+ np.exp(-z))
        dz = A - Y_train
        dw = np.dot(X_train.T, dz) / m
        db = np.sum(dz) / m    
        W -= learningRate * dw
        b -= learningRate * db
    return W, b
